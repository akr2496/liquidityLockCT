public int maximumGap(final int[] arr) {
        int n = arr.length;
        int[] rightMax = new int[n];
        rightMax[n-1] = arr[n-1];
        for(int i = n-2; i >= 0; i--)
            rightMax[i] = Math.max(rightMax[i+1] , arr[i]);
   
        int maxDist = Integer.MIN_VALUE;
        int i = 0, j = 0;
        while(i < n && j < n){
            if(rightMax[j] >= arr[i]){
                maxDist = Math.max( maxDist, j-i );
                j++;
            }
            else i++;
        }    
        return maxDist;
    }
